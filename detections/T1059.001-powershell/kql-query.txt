// Kusto KQL — Microsoft Sentinel / Google SecOps (Chronicle)
// T1059.001 — PowerShell Encoded Command Execution
//
// For Microsoft Sentinel: run in Log Analytics workspace > Logs, or create a Scheduled Alert rule.
// For Google SecOps/Chronicle: the platform uses a Kusto-like dialect — syntax is compatible
// but table names differ (use UDM or raw log tables as appropriate for your instance).
//
// Table: DeviceProcessEvents (Microsoft Defender for Endpoint via Sentinel)
// Adjust table name if ingesting Sysmon via Windows Security Events or a third-party connector.

DeviceProcessEvents
| where TimeGenerated >= ago(24h)

// Match powershell.exe or pwsh.exe
| where FileName in~ ("powershell.exe", "pwsh.exe")

// Match -EncodedCommand and common abbreviations (-enc, -ec, -e) followed by base64 payload
// Kusto regex is case-sensitive by default; (?i) enables case-insensitive matching
| where ProcessCommandLine matches regex @"(?i)[\s\-](e(nc(odedcommand)?|c))\s+[A-Za-z0-9+/]{20,}={0,2}"

// Exclude known-legitimate parent processes
| where InitiatingProcessFileName !in~ (
    "MpCmdRun.exe",
    "SenseIR.exe",
    "MsMpEng.exe",
    "ccmexec.exe",
    "UsoClient.exe",
    "WUDFHost.exe"
)

// Project relevant fields for triage
| project
    TimeGenerated,
    DeviceName,
    AccountName,
    FileName,
    ProcessCommandLine,
    InitiatingProcessFileName,
    InitiatingProcessCommandLine,
    SHA256,
    FolderPath

| order by TimeGenerated desc
